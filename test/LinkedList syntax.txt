public class Test2 {
    public static void main(String[] args){
        Node headNode = null;
        Node tempNode = headNode;
        for(int i = 0; i < 100; i++){
            if(headNode == null){
                tempNode = new Node();
                headNode = tempNode;
            }
            else{
                tempNode.nextNode = new Node();
                tempNode = tempNode.nextNode;

            }
            tempNode.data = i+1000;
        }
        tempNode = headNode;
        while(tempNode != null){
            System.out.println(tempNode.data);
            tempNode = tempNode.nextNode;
        }
    }
}
class Node{
    public Node nextNode;
    public int data;

}
